__global__ void mmul(const float* A, const float* B, float* C, int dy, int dx) {
	// m * k Âíå k * n  Áü©Èòµ‰πòÊ≥ïÁöÑÈÄöÁî®ÂÜôÊ≥ï
	int m = dy, k = dx, n = dy;
	// ËÆ°ÁÆóÂΩìÂâçÁ∫øÁ®ãÂØπÂ∫îÁöÑË°åÂíåÂàó

	// ÂèØ‰ª•ËÆ§‰∏∫ÔºåÂè™Áî®rowÂíåcolËÉΩÁî®Êù•Ë°®Á§∫C‰∏≠ÂÖÉÁ¥†ÁöÑÁ∫øÁ®ãÊâç‰ºöËøõË°åËÆ°ÁÆó
	// ‰∏ç‰ºöÂèëÁîüÁ∫øÁ®ãÁöÑÂÜó‰ΩôËÆ°ÁÆó
	int row = blockIdx.y * blockDim.y + threadIdx.y;
	int col = blockIdx.x * blockDim.x + threadIdx.x;

	// C[row][col] ‰∏∫ AÁöÑÁ¨¨rowË°å * BÁöÑÁ¨¨colÂàó
	if (row < m && col < n) {
		float sum = 0.0f;
		for (int i = 0; i < k; i++) {
			sum += A[row * k + i] * B[i * n + col]; // A Ë°å‰∏ªÂ∫èÔºåB Âàó‰∏ªÂ∫è
		}
		C[row * n + col] = sum; // C Ë°å‰∏ªÂ∫è
	}
}
