    int on = 1;

        perror("setsockopt");

void EventLoop::WaitEpollFd() {

    } while(-1 == nready && errno == EINTR);

        std::cout << ">>epoll_wait timeout" << std::endl;
    } else {
        //ÂèØ‰ª•Âà§Êñ≠‰∏Ä‰∏ãÔºåÊñá‰ª∂ÊèèËø∞Á¨¶ÊòØ‰∏çÊòØÂ∑≤ÁªèËææÂà∞‰∫Ü1024

            int fd = evtList_[idx].data.fd;
            //Êü•ÁúãÊñá‰ª∂ÊèèËø∞Á¨¶ÊòØ‰∏çÊòØlistenfd

                    //Â§ÑÁêÜÊñ∞ÁöÑËøûÊé•

                }
            } else if (fd == evtfd_) {            //ËØ¥ÊòéÊúâËøûÊé•ÁöÑ‰ªªÂä°Â§ÑÁêÜÂÆåÊàê‰∫Ü

    std::unique_lock lock(mutex_);
    while(IsFull()) {
        //Á≠âÂæÖnot_full_Êù•Âî§ÈÜí

    std::unique_lock lock(mutex_);
    while(IsEmpty() && flag_) {
        //Á≠âÂæÖnot_empty_Êù•Âî§ÈÜí

    } else {

    }

}

    flag_ = false;
    not_empty_.notify_all();
}
